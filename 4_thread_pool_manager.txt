#include <iostream>
#include <vector>
#include <queue>
#include <thread>
#include <mutex>
#include <condition_variable>
#include <functional>

using namespace std;

// --- ThreadPool Class ---
class ThreadPool {
private:
    // Synchronization Primitives
    mutex queue_mutex;                  // Protects the shared task_queue
    condition_variable condition;        // Signals workers when a task is available
    bool stop_flag = false;             // Flag to signal workers to exit

    // Thread Pool Resources
    vector<thread> workers;             // The collection of worker threads
    queue<function<void()>> tasks;      // The queue of tasks (functions) to be executed

    // Worker function: runs in each thread
    void worker_loop() {
        while (true) {
            function<void()> task; // Task function to execute

            { // --- START CRITICAL SECTION ---
                unique_lock<mutex> lock(queue_mutex);

                // Wait: Block the worker thread until the queue is NOT empty OR the stop flag is set
                // The condition variable acts like a semaphore, signaling when a task is ready.
                condition.wait(lock, [this] {
                    return stop_flag || !tasks.empty();
                });

                // If stop_flag is true AND the queue is empty, the worker exits gracefully.
                if (stop_flag && tasks.empty()) {
                    return;
                }

                // Retrieve the next task from the queue
                task = move(tasks.front());
                tasks.pop();

            } // --- END CRITICAL SECTION: Lock is automatically released here ---

            // Execute the task OUTSIDE the lock
            task(); 
        }
    }

public:
    // Constructor: Initializes the pool with a specified number of threads
    ThreadPool(size_t num_threads) {
        // Create (launch) the worker threads
        for (size_t i = 0; i < num_threads; ++i) {
            workers.emplace_back(&ThreadPool::worker_loop, this);
            cout << "Main: Launched worker thread " << i << endl;
        }
    }

    // Destructor: Safely shuts down the pool
    ~ThreadPool() {
        { // Acquire lock to set the stop flag and notify all workers
            unique_lock<mutex> lock(queue_mutex);
            stop_flag = true;
        } 
        
        // Notify ALL waiting threads to check the stop_flag and exit
        condition.notify_all();

        // Wait for all worker threads to finish joining
        for (thread& worker : workers) {
            if (worker.joinable()) {
                worker.join();
            }
        }
        cout << "Main: Thread pool safely shut down." << endl;
    }

    // Function to add a task to the queue
    template<class F>
    void addTask(F&& task) {
        { // Acquire lock to modify the shared task queue
            unique_lock<mutex> lock(queue_mutex);
            // Add the task (packaged as std::function) to the queue
            tasks.emplace(forward<F>(task));
        }
        
        // Signal one waiting worker that a new task is available
        condition.notify_one(); 
    }
};


// --- Example Task Function ---
void execute_task(int id, int duration_ms) {
    cout << "  [Worker " << this_thread::get_id() << "] Starting Task " << id << " for " << duration_ms << "ms." << endl;
    this_thread::sleep_for(chrono::milliseconds(duration_ms));
    cout << "  [Worker " << this_thread::get_id() << "] Finished Task " << id << "." << endl;
}

// --- Main Program to Test the Pool ---
int main() {
    // 1. Initialize a thread pool with 4 worker threads
    ThreadPool pool(4);
    cout << "--------------------------------------------------------" << endl;

    // 2. Add 15 tasks (more tasks than threads) to the queue
    int num_tasks = 15;
    cout << "Main: Adding " << num_tasks << " tasks to the pool queue..." << endl;
    
    for (int i = 1; i <= num_tasks; ++i) {
        // Create a task function (lambda) and add it to the pool
        int duration = (i % 3 == 0) ? 1000 : 300; // Some tasks take longer
        pool.addTask([i, duration] { 
            execute_task(i, duration); 
        });
    }

    cout << "Main: Finished adding tasks. Waiting for workers to finish..." << endl;
    // The main function exits and the pool's destructor is automatically called, 
    // which safely waits for all tasks and threads to complete.

    return 0;
}
