#include <iostream>
#include <vector>
#include <thread>
#include <mutex>
#include <condition_variable>
#include <chrono>

using namespace std;

// --- Shared Resources ---
const int MAX_BUFFER_SIZE = 5;
vector<int> buffer;
int item_counter = 0; // Global counter for items produced

// --- Synchronization Primitives ---
mutex mtx; // Mutex: Protects access to the shared buffer
condition_variable cv_producer; // Producer waits on this (when buffer is full)
condition_variable cv_consumer; // Consumer waits on this (when buffer is empty)

// --- Simulating Semaphores using C++ Primitives ---
// Counting Semaphores are generally used to limit resources.
// In this case, the buffer size and item count serve as the resource limits,
// which is managed by the condition variables (wait/signal).

// --- Functions ---

void producer() {
    for (int i = 0; i < 10; ++i) { // Produce 10 items
        unique_lock<mutex> lock(mtx); // Acquire Lock (Mutual Exclusion)

        // CONDITION: Wait while the buffer is full (equivalent to waiting on a 'empty slots' semaphore count of 0)
        cv_producer.wait(lock, []{ return buffer.size() < MAX_BUFFER_SIZE; });

        // CRITICAL SECTION START
        int produced_item = ++item_counter;
        buffer.push_back(produced_item);
        
        cout << "P: Produced item " << produced_item << ". Buffer size: " << buffer.size() << endl;
        // CRITICAL SECTION END

        lock.unlock(); // Release Lock

        // SIGNAL: Notify any waiting consumer threads that an item is available
        cv_consumer.notify_one(); 

        // Simulate work time
        this_thread::sleep_for(chrono::milliseconds(200));
    }
}

void consumer() {
    for (int i = 0; i < 10; ++i) { // Consume 10 items
        unique_lock<mutex> lock(mtx); // Acquire Lock (Mutual Exclusion)

        // CONDITION: Wait while the buffer is empty (equivalent to waiting on a 'filled slots' semaphore count of 0)
        cv_consumer.wait(lock, []{ return !buffer.empty(); });

        // CRITICAL SECTION START
        int consumed_item = buffer.back();
        buffer.pop_back();

        cout << "C: Consumed item " << consumed_item << ". Buffer size: " << buffer.size() << endl;
        // CRITICAL SECTION END

        lock.unlock(); // Release Lock

        // SIGNAL: Notify any waiting producer threads that space is available
        cv_producer.notify_one();

        // Simulate work time
        this_thread::sleep_for(chrono::milliseconds(500));
    }
}

int main() {
    cout << "--- Producer-Consumer Synchronization Demo ---" << endl;
    cout << "Buffer Max Size: " << MAX_BUFFER_SIZE << endl;
    cout << "Goal: Produce and consume 10 items safely." << endl;
    
    // Create threads
    thread producer_thread(producer);
    thread consumer_thread(consumer);

    // Wait for threads to finish
    producer_thread.join();
    consumer_thread.join();

    cout << "\n--- Demo Finished ---" << endl;
    cout << "Final item count produced: " << item_counter << endl;
    cout << "Final buffer size: " << buffer.size() << endl;

    return 0;
}
